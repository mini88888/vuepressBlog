> Hook是React16.8的新增特性。可以让你不编写class的情况下使用state以及其他的React特性

### 可以用两个例子认识Hook
```js
import React, { useState } from 'react';

function Example() {
  // 声明一个叫 "count" 的 state 变量
  const [count, setCount] = useState(0);

    <div>
      <p>You clicked {count} times</p>
      <button onClick={() => setCount(count + 1)}>
        Click me
      </button>
    </div>
  );
}
```

  }
